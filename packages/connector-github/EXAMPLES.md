## Basic Setup

### Installation and Authentication

```bash
# Install in your HoloScript project
pnpm add @holoscript/connector-github

# Set GitHub token
export GITHUB_TOKEN="ghp_your_token_here"
```

```typescript
import { GitHubConnector } from '@holoscript/connector-github';

async function main() {
  const connector = new GitHubConnector();

  try {
    await connector.connect();
    console.log('✅ Connected to GitHub');

    const healthy = await connector.health();
    console.log('Health:', healthy);

    const tools = await connector.listTools();
    console.log(`${tools.length} MCP tools available`);
  } finally {
    await connector.disconnect();
  }
}

main();
```

## CI/CD Integration

### Automated Scene Compilation on PR

```typescript
import { GitHubConnector } from '@holoscript/connector-github';

async function compilePRScenes(owner: string, repo: string, prNumber: number) {
  const connector = new GitHubConnector();
  await connector.connect();

  // Get PR information
  const pr = (await connector.executeTool('github_pr_get', {
    owner,
    repo,
    pull_number: prNumber,
  })) as any;

  const branch = pr.data.head.ref;

  // Compile all .holo files in the PR branch
  const result = (await connector.executeTool('github_holoscript_compile_preview', {
    owner,
    repo,
    ref: branch,
  })) as any;

  // Post preview links as PR comment
  if (result.summary.successful > 0) {
    const previewLinks = result.files
      .filter((f: any) => f.success)
      .map((f: any) => `- [${f.file}](${f.previewUrl})`)
      .join('\n');

    await connector.executeTool('github_pr_comment', {
      owner,
      repo,
      pull_number: prNumber,
      body: `## 🎨 Preview Links\n\n${previewLinks}\n\n---\n*Generated by HoloScript Connector*`,
    });
  }

  await connector.disconnect();
  return result;
}

// Usage
compilePRScenes('holoscript', 'holoscript', 456);
```

### Daily Scene Validation

```typescript
import { GitHubConnector } from '@holoscript/connector-github';

async function validateAllScenes(owner: string, repo: string) {
  const connector = new GitHubConnector();
  await connector.connect();

  // Validate all scenes on main branch
  const validation = (await connector.executeTool('github_holoscript_validate_scene', {
    owner,
    repo,
    ref: 'main',
  })) as any;

  // Create issue if errors found
  if (validation.summary.invalid > 0) {
    const errorReport = validation.files
      .filter((f: any) => !f.valid)
      .map((f: any) => `### ${f.file}\n${f.errors.map((e: string) => `- ${e}`).join('\n')}`)
      .join('\n\n');

    await connector.executeTool('github_issue_create', {
      owner,
      repo,
      title: `🚨 Scene Validation Errors Detected`,
      body: `Automated validation found errors in HoloScript scenes:\n\n${errorReport}`,
      labels: ['bug', 'automated'],
    });
  }

  await connector.disconnect();
  return validation;
}
```

## Automated PR Workflow

### Complete Feature Development Flow

```typescript
import { GitHubConnector } from '@holoscript/connector-github';
import { execSync } from 'child_process';

async function createFeaturePR(
  owner: string,
  repo: string,
  featureName: string,
  sceneFile: string
) {
  const connector = new GitHubConnector();
  await connector.connect();

  const branchName = `feature/${featureName}`;

  try {
    // 1. Validate scene locally first
    const validation = (await connector.executeTool('github_holoscript_validate_scene', {
      owner,
      repo,
      ref: branchName,
    })) as any;

    if (validation.summary.invalid > 0) {
      console.error('❌ Validation failed, fix errors before creating PR');
      return;
    }

    // 2. Create pull request
    const pr = (await connector.executeTool('github_pr_create', {
      owner,
      repo,
      title: `Add ${featureName} feature`,
      body: `## 🎨 New Feature: ${featureName}\n\nThis PR adds a new holographic scene.`,
      head: branchName,
      base: 'main',
    })) as any;

    console.log(`✅ PR created: #${pr.data.number}`);

    // 3. Compile preview
    const preview = (await connector.executeTool('github_holoscript_compile_preview', {
      owner,
      repo,
      ref: branchName,
      files: [sceneFile],
    })) as any;

    // 4. Add preview comment
    if (preview.files[0]?.previewUrl) {
      await connector.executeTool('github_pr_comment', {
        owner,
        repo,
        pull_number: pr.data.number,
        body: `## 🎨 3D Preview\n\n[View Live Preview](${preview.files[0].previewUrl})`,
      });
    }

    // 5. Request review from team
    // (Would use GitHub's request reviewers API)

    return pr.data;
  } finally {
    await connector.disconnect();
  }
}

// Usage
createFeaturePR('myorg', 'vr-gallery', 'water-reflection', 'scenes/water.holo');
```

## Scene Validation Bot

### Automated Review Bot

```typescript
import { GitHubConnector } from '@holoscript/connector-github';

class HoloScriptReviewBot {
  private connector: GitHubConnector;

  constructor() {
    this.connector = new GitHubConnector();
  }

  async reviewPR(owner: string, repo: string, prNumber: number) {
    await this.connector.connect();

    try {
      // Get PR details
      const pr = (await this.connector.executeTool('github_pr_get', {
        owner,
        repo,
        pull_number: prNumber,
      })) as any;

      const branch = pr.data.head.ref;

      // Validate scenes
      const validation = (await this.connector.executeTool('github_holoscript_validate_scene', {
        owner,
        repo,
        ref: branch,
      })) as any;

      // Compile previews
      const compilation = (await this.connector.executeTool('github_holoscript_compile_preview', {
        owner,
        repo,
        ref: branch,
      })) as any;

      // Determine review decision
      const hasErrors = validation.summary.invalid > 0;
      const hasWarnings = validation.summary.totalWarnings > 0;
      const allCompiled = compilation.summary.failed === 0;

      let reviewEvent: 'APPROVE' | 'REQUEST_CHANGES' | 'COMMENT';
      let reviewBody = '## 🤖 Automated Review\n\n';

      if (hasErrors || !allCompiled) {
        reviewEvent = 'REQUEST_CHANGES';
        reviewBody += '### ❌ Changes Requested\n\n';
        reviewBody += `- ${validation.summary.totalErrors} validation errors\n`;
        reviewBody += `- ${compilation.summary.failed} compilation failures\n\n`;
        reviewBody += 'Please fix the errors before merging.';
      } else if (hasWarnings) {
        reviewEvent = 'COMMENT';
        reviewBody += '### ⚠️ Warnings Found\n\n';
        reviewBody += `- ${validation.summary.totalWarnings} warnings detected\n\n`;
        reviewBody += 'Consider addressing warnings for best practices.';
      } else {
        reviewEvent = 'APPROVE';
        reviewBody += '### ✅ All Checks Passed\n\n';
        reviewBody += '- All scenes validated successfully\n';
        reviewBody += '- All compilations succeeded\n';
      }

      // Submit review
      await this.connector.executeTool('github_pr_review', {
        owner,
        repo,
        pull_number: prNumber,
        body: reviewBody,
        event: reviewEvent,
      });

      console.log(`✅ Review submitted: ${reviewEvent}`);
    } finally {
      await this.connector.disconnect();
    }
  }
}

// Usage
const bot = new HoloScriptReviewBot();
bot.reviewPR('holoscript', 'holoscript', 789);
```

## Advanced: Workflow Orchestration

### GitHub Actions + MCP Integration

```typescript
import { GitHubConnector } from '@holoscript/connector-github';

async function orchestrateWorkflow(owner: string, repo: string) {
  const connector = new GitHubConnector();
  await connector.connect();

  // Trigger the HoloScript CI workflow
  await connector.executeTool('github_workflow_run', {
    owner,
    repo,
    workflow_id: 'holoscript-ci.yml',
    ref: 'main',
    inputs: {
      target: 'r3f',
      validate: 'true',
      deploy: 'preview',
    },
  });

  console.log('✅ Workflow triggered');

  // Poll for completion
  await new Promise((resolve) => setTimeout(resolve, 5000));

  // Check workflow runs
  const runs = (await connector.executeTool('github_workflow_runs_list', {
    owner,
    repo,
    status: 'completed',
  })) as any;

  console.log(`Latest run: ${runs.data.workflow_runs[0]?.conclusion}`);

  await connector.disconnect();
}
```

## Best Practices

1. **Always disconnect** - Call `disconnect()` in a `finally` block
2. **Type safety** - Cast `executeTool()` results with `as any` or define interfaces
3. **Error handling** - Wrap operations in try-catch for production use
4. **Rate limiting** - GitHub API has rate limits (5000 req/hour for authenticated)
5. **Webhook integration** - For production bots, use GitHub webhooks instead of polling
